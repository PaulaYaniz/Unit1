### Multiplication table

```.py
def mulTable():

    number = int(input("Please enter an integer between 2 and 10: "))

    while number < 2 or number > 10:
        number = int(input("ERROR. Please enter an integer between 2 and 10: "))

    out = ""
    for i in range(1, 11):
        result = number*i
        out += (f"{number} x {i} = {result}\n")

    print(out)


# Test the function
mulTable()
```
