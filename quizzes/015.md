def open_doors(num_students: int):
    door_list = []
    total_open = 0

    for door in range(num_students):
        door_list.append(False)

    for student in range(1, num_students + 1):
        for door in range(1, num_students + 1):
            if door % student == 0:
                door_list[door-1] = not door_list[door-1]

    for count in range(num_students):
        if door_list[count]:
            total_open += 1

    return total_open

# Test function
out = open_doors(num_students=int(input("Please enter the number of students and doors: ")))
print(f"There are {out} open doors.")
